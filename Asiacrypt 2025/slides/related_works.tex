\begin{frame}
    \frametitle{Related Works}
    \begin{itemize}
        \item Structure-aware PSI was proposed by \cite{garimella2022structure} for $L_\infty$ metric, introducing the framework:
        \begin{center}
            Spatial Hashing $\to$ Function Secret Sharing $\to$ Matching        
        \end{center}
        \pause
        \item \cite{garimella2023malicious} upgrades the protocol to malicous security.
        \item \cite{garimella2024computation} improve spatial hashing, and introduces trade-offs between computation and communication.
        \pause
        \item \cite{van2024fuzzy,gao2024efficient} generalizes the approach into \textit{fuzzy map}, using Homomorphic Encryption, and support more distance metrics: $L_p$ and Hamming distance.
        \item \cite{zhang2025fast,van2025fuzzy} replaces HE with symmetric key techniques, \cite{piske2025distance} introduces distance-aware OT, to improve fuzzy maps efficiency.
    \end{itemize}
\end{frame}
