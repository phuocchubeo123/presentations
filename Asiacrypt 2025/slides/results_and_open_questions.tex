    \only<2>{\frametitle{Complexity Analysis and Open Questions}}
    \begin{itemize}
        \item Distributed DCF:
        \begin{itemize}
            \item Computation: $O(\delta \cdot d \cdot N_A)$ hashes. \visible<2,3>{{\color{red}$\rightarrow O(\log \delta \cdot d \cdot N_A)$?}}
            \item Communication: $O(\log \delta \cdot \kappa \cdot d \cdot N_A)$ bits.
        \end{itemize}
        \item Tree search: Many trade-offs in different settings.
        \begin{enumerate}
            \item LAN:
            \begin{itemize}
                \item Bob sends $O((\log \delta)^d \cdot N_B)$ hashes. \visible<3>{{\color{red}$\rightarrow$ linear in $d$?}}
                \item Alice searches with $O(\left(\frac{\log \delta}{2}\right)^d \cdot N_A)$ critical prefixes and $O(\log \delta \cdot N_A)$ search steps.
            \end{itemize}
            \item WAN:
            \begin{itemize}
                \item Bob sends $O(N_B)$ hashes.
                \item Alice searches the whole space: $O((\log \delta)^d \cdot N_A)$ hashes. \visible<3>{{\color{red} $\rightarrow$ linear in $d$?}}
            \end{itemize}
            \item One example of customized trade-off:
            \begin{itemize}
                \item Bob sends $O(\left(\frac{\log \delta}{2}\right)^d \cdot N_B)$ hashes. \visible<3>{{\color{red}$\rightarrow$ linear in $d$?}}
                \item Alice searches critical prefixes with $O(4 \cdot \left(\frac{\log \delta}{2}\right)^d \cdot N_A)$ hashes and $O(4 \cdot d \cdot \log \delta)$ search steps.
            \end{itemize}
        \end{enumerate}
    \end{itemize}
\end{frame}
